### 12bit ADC 값을 3.3V 의 Rate(0 ~ 1.0)로 return
'''c
float getADCRate(uint16_t adc_12bit_code)
{
	float fret = 0.0;
	fret = (float)(adc_12bit_code*16) * (1.0f / (float)0xFFFF);
	return fret;
}
'''
